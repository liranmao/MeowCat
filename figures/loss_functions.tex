\section*{Loss Functions}

\textbf{Visium loss} (MSE on soft cell-type proportions from RCTD):
\begin{equation}
\loss_{\text{vis}} = \frac{1}{N_v} \sum_{i \in \mathcal{V}} \left\| \hat{p}_i - y_i \right\|^2
\end{equation}

\textbf{Xenium loss} (cross-entropy on hard single-cell labels):
\begin{equation}
\loss_{\text{xen}} = -\frac{1}{N_x} \sum_{i \in \mathcal{X}} \log \hat{p}_{i,\, c_i}
\end{equation}

\textbf{CDAN domain adversarial loss} with entropy conditioning:
\begin{equation}
\loss_{\text{dom}} = -\frac{1}{N} \sum_{i=1}^{N} w_i \sum_{d=1}^{D} \mathbf{1}[d_i = d]\, \log \hat{d}_i^{(d)}, \quad w_i = 1 + \exp(-H_i)
\end{equation}
where the domain classifier receives CDAN features:
\begin{equation}
f_i = \frac{\hat{p}_i \otimes z_i}{\sqrt{D}}
\end{equation}
and a gradient reversal layer (GRL) negates gradients during backpropagation:
\begin{equation}
\text{GRL}(x) = x \ (\text{forward}), \qquad \nabla \rightarrow -\lambda\, \nabla \ (\text{backward})
\end{equation}

\textbf{Reconstruction loss} (masked MSE):
\begin{equation}
\loss_{\text{rec}} = \frac{1}{|\mathcal{M}|} \sum_{j \in \mathcal{M}} \left\| \hat{x}_j - x_j \right\|^2, \quad \mathcal{M} \sim \text{Bernoulli}(r)
\end{equation}

\textbf{Total loss:}
\begin{equation}
\loss = \loss_{\text{vis}} + \alpha\, \loss_{\text{xen}} + \lambda(t)\, \loss_{\text{dom}} + \rho\, \loss_{\text{rec}}
\end{equation}

\textbf{Adversarial schedule:}
\begin{equation}
\lambda(t) = \lambda_{\max} \left( \frac{2}{1 + e^{-10p}} - 1 \right), \quad p = \frac{\text{epoch}}{\text{max\_epochs}}
\end{equation}

\section*{Multi-Phase Training}

\begin{table}[h]
\centering
\begin{tabular}{@{}lllll@{}}
\toprule
\textbf{Phase}  & \textbf{Aim} & \textbf{Data} & \textbf{Objective} & \textbf{Encoder Freezing} \\
\midrule
Phase 0 & Reconstruction pretraining & All samples    & $\loss_{\text{rec}}$ only             & None (full encoder trains) \\
Phase 1 & Spot-data weak supervision & Spot-data only    & $\loss_{\text{vis}} + \loss_{\text{dom}} + \loss_{\text{rec}}$ & First $N$ layers frozen \\
Phase 2 & Cell-data fine-tuning & Cell-data only    & $\loss_{\text{xen}} + \loss_{\text{dom}} + \loss_{\text{rec}}$ & First $N{-}1$ layers frozen \\
\bottomrule
\end{tabular}
\end{table}

\begin{description}[style=unboxed, leftmargin=0pt]
\item[Phase 0 --- Reconstruction pretraining.]
Learns a general-purpose latent space from all H\&E embeddings without any labels. The masked autoencoder objective forces the encoder to capture tissue morphology before seeing any supervision signal.

\item[Phase 1 --- Visium weak supervision.]
Trains on Visium spots with soft cell-type proportions from RCTD deconvolution (${\sim}55\,\mu$m resolution). Partial encoder freezing preserves pretrained representations in early layers while adapting later layers.

\item[Phase 2 --- Xenium fine-tuning.]
Fine-tunes on Xenium single-cell annotations with hard labels at cellular resolution. Fewer layers are frozen ($N{-}1$ vs $N$) to allow slightly more adaptation, but the largely frozen encoder prevents catastrophic forgetting of Visium-learned features. CDAN domain adversarial loss aligns Visium and Xenium feature spaces across all phases.
\end{description}
